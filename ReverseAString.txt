// 1st approach | using 2 pointers and swapping
// time complexity :- O(N)
// space complexity :- O(1)

#include<iostream>
using namespace std;

int main() {
    
    string s;
    cout<<"Enter a string :- ";
    cin>>s;
    int i = 0, j = s.size() - 1;
    while(i <= j) {
        swap(s[i], s[j]);
        i++;
        j--;
    }
    cout<<"Reverse of "<<s<<" is :- ";
    for(int i = 0; i < s.size(); i++) {
        cout<<s[i];
    }
    
    return 0;
}
