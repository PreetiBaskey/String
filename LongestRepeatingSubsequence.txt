#include<iostream>
using namespace std;

int getLCS(string s1, string s2, int i, int j) {
    int m = s1.size();
    int n = s2.size();
    
    if(i == m || j == n) {
        return 0;
    }
    if(s1[i] == s2[j] && i != j) {
        return 1 + getLCS(s1, s2, i + 1, j + 1);
    }
    else {
        int c1 = getLCS(s1, s2, i, j + 1);
        int c2 = getLCS(s1, s2, i + 1, j);
        return max(c1, c2);
    }
}

int main() {
    
    string s;
    cout<<"Enter a string :- ";
    cin>>s;
    int LCS;
    LCS = getLCS(s, s, 0, 0);
    cout<<"Longest common subsequence is :- "<<LCS;
    
    return 0;
}
