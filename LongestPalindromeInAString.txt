#include<iostream>
#include<climits>
using namespace std;

bool solve(string &s, int i, int j) {
    if(i >= j) {
        return true;
    }
    if(s[i] == s[j]) {
        return solve(s, i + 1, j - 1);
    }
    else {
        return false;
    }
}

string findLongestPalindrome(string &s) {
    int n = s.size();
    int maxLength = INT_MIN;
    int sp = 0;
    for(int i = 0; i < n; i++) {
        for(int j = i; j < n; j++) {
            if(solve(s, i, j) == true) {
                if(j - i + 1 > maxLength) {
                    maxLength = j - i + 1;
                    sp = i;
                }
            }
        }
    }
    
    return s.substr(sp, maxLength);
}

int main() {
    
    string s, result;
    cout<<"Enter a string :- ";
    cin>>s;
    result = findLongestPalindrome(s);
    cout<<"Longest palindrome in "<<s<<" is :- "<<result;
    
    return 0;
}
