// 1st approach | using unordered_map
// time complexity :- O(N)
// space complexity :- O(N)


#include<iostream>
#include<unordered_map>
using namespace std;

int main() {
    
    string s;
    cout<<"Enter a string :- ";
    cin>>s;
    
    int n = s.size();
    unordered_map<char, int> mp;
    for(int i = 0; i < n; i++) {
        mp[s[i]]++;
    }
    cout<<"Duplicate characters are :"<<endl;
    for(auto &data: mp) {
        if(data.second > 1) {
            cout<<data.first<<" :- "<<data.second<<endl;
        }
    }
    
    return 0;
}
